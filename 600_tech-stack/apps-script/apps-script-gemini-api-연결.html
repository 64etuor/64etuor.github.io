function columnLetterToIndex(columnLetter) {
  if (!columnLetter || typeof columnLetter !== 'string') return -1;
  const upperCaseLetter = columnLetter.toUpperCase();
  let index = 0;
  for (let i = 0; i < upperCaseLetter.length; i++) {
    const charCode = upperCaseLetter.charCodeAt(i);
    if (charCode < 'A'.charCodeAt(0) || charCode > 'Z'.charCodeAt(0)) return -1; 
    index = index * 26 + (charCode - 'A'.charCodeAt(0) + 1);
  }
  return index - 1; // 0-based index
}

function triggerGeminiUpdate() {
  // --- Configuration ---
  const TASK_SHEET_NAME = "task"; // 작업 데이터가 있는 시트 이름
  const ADVICE_SHEET_NAME = "advice"; // 결과(조언)를 저장할 시트 이름
  const PROMPT_SHEET_NAME = "ai_prompt"; // 프롬프트가 저장된 시트 이름
  const PROMPT_CELL_SECRETARY = "A2"; // 비서 프롬프트 셀 위치
  const PROMPT_CELL_NUTRITIONIST = "B2"; // 영양사 프롬프트 셀 위치
  const TASK_DATA_RANGE = "D:J"; // task 시트에서 가져올 데이터 열 범위 (예: "D:J")
  const DATE_COLUMN_LETTER = 'E'; // 날짜가 있는 실제 열 문자
  const DAYS_TO_LOOK_BACK = 7; // 며칠 전 데이터까지 볼 것인지 설정 (7일)
  const GEMINI_MODEL = "gemini-1.5-flash-latest"; // 사용할 Gemini 모델
  const MAX_ROWS_TO_CHECK = 9999; // 확인할 최대 행 수 (성능 고려)
  // --------------------

  // --- API 키 가져오기 (Properties Service 사용) ---
  const scriptProperties = PropertiesService.getScriptProperties();
  const GEMINI_API_KEY = scriptProperties.getProperty('GEMINI_API_KEY');
  if (!GEMINI_API_KEY) {
    Logger.log("스크립트 속성에서 'GEMINI_API_KEY'를 찾을 수 없습니다. 프로젝트 설정 > 스크립트 속성을 확인하세요.");
    SpreadsheetApp.getUi().alert("오류: API 키가 설정되지 않았습니다. 스크립트 속성을 확인하세요.");
    return; // API 키가 없으면 함수 실행 중단
  }
  // ---------------------------------------------

  const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
  const taskSheet = spreadsheet.getSheetByName(TASK_SHEET_NAME);
  const adviceSheet = spreadsheet.getSheetByName(ADVICE_SHEET_NAME);
  const promptSheet = spreadsheet.getSheetByName(PROMPT_SHEET_NAME);

  // 시트 존재 여부 확인
  if (!taskSheet) {
    Logger.log(`시트 "${TASK_SHEET_NAME}"을(를) 찾을 수 없습니다.`);
    SpreadsheetApp.getUi().alert(`시트 "${TASK_SHEET_NAME}"을(를) 찾을 수 없습니다.`);
    return;
  }
  if (!adviceSheet) {
    Logger.log(`시트 "${ADVICE_SHEET_NAME}"을(를) 찾을 수 없습니다.`);
    SpreadsheetApp.getUi().alert(`시트 "${ADVICE_SHEET_NAME}"을(를) 찾을 수 없습니다.`);
    return;
  }
  if (!promptSheet) {
    Logger.log(`시트 "${PROMPT_SHEET_NAME}"을(를) 찾을 수 없습니다.`);
    SpreadsheetApp.getUi().alert(`시트 "${PROMPT_SHEET_NAME}"을(를) 찾을 수 없습니다.`);
    return;
  }

  // --- 날짜 계산 ---
  const now = new Date();
  const timeZone = spreadsheet.getSpreadsheetTimeZone();
  const currentDate = Utilities.formatDate(now, timeZone, "yyyy-MM-dd");
  const pastDate = new Date(now);
  pastDate.setDate(now.getDate() - DAYS_TO_LOOK_BACK);
  pastDate.setHours(0, 0, 0, 0); // 시간, 분, 초, 밀리초를 0으로 설정하여 날짜만 비교
  const pastDateTimestamp = pastDate.getTime();
  Logger.log(`오늘 날짜: ${currentDate}`);
  Logger.log(`${DAYS_TO_LOOK_BACK}일 전 날짜 기준 (포함): ${Utilities.formatDate(pastDate, timeZone, "yyyy-MM-dd")}`);

  // --- 프롬프트 가져오기 ---
  const secretaryPrompt = promptSheet.getRange(PROMPT_CELL_SECRETARY).getValue();
  const nutritionistPrompt = promptSheet.getRange(PROMPT_CELL_NUTRITIONIST).getValue();
  if (!secretaryPrompt || !nutritionistPrompt) {
    Logger.log("프롬프트 시트에서 프롬프트를 가져올 수 없습니다. A2 또는 B2 셀을 확인하세요.");
    SpreadsheetApp.getUi().alert("프롬프트 시트에서 프롬프트를 가져올 수 없습니다. A2 또는 B2 셀을 확인하세요.");
    return;
  }
  Logger.log(`비서 프롬프트 로드 완료.`);
  Logger.log(`영양사 프롬프트 로드 완료.`);

  // --- Task 데이터 가져오기 및 필터링 ---
  const rangeParts = TASK_DATA_RANGE.split(':');
  if (rangeParts.length !== 2) {
     Logger.log(`오류: TASK_DATA_RANGE 형식("${TASK_DATA_RANGE}")이 잘못되었습니다. "D:J"와 같은 형식이어야 합니다.`);
     SpreadsheetApp.getUi().alert(`오류: 데이터 범위 형식이 잘못되었습니다. ("${TASK_DATA_RANGE}")`);
     return;
  }
  const startColumnLetter = rangeParts[0];
  const endColumnLetter = rangeParts[1];
  const rangeA1Notation = startColumnLetter + "1:" + endColumnLetter + MAX_ROWS_TO_CHECK;

  // 날짜 열 인덱스 계산
  const startColumnIndex = columnLetterToIndex(startColumnLetter);       // 예: D -> 3
  const endColumnIndex = columnLetterToIndex(endColumnLetter);         // 예: J -> 9
  const dateColumnIndexAbsolute = columnLetterToIndex(DATE_COLUMN_LETTER); // 예: E -> 4

  if (startColumnIndex === -1 || endColumnIndex === -1 || dateColumnIndexAbsolute === -1) {
      Logger.log(`오류: 설정된 열 문자(시작:${startColumnLetter}, 끝:${endColumnLetter}, 날짜:${DATE_COLUMN_LETTER}) 중 잘못된 값이 있습니다.`);
      SpreadsheetApp.getUi().alert(`오류: 설정된 열 문자가 잘못되었습니다.`);
      return;
  }

  // 가져온 데이터 배열 내에서의 날짜 열 인덱스 (0부터 시작)
  const dateColumnIndexInRange = dateColumnIndexAbsolute - startColumnIndex; // 예: 4 - 3 = 1

  // 날짜 열이 가져오는 범위 내에 있는지 확인
  if (dateColumnIndexInRange < 0 || dateColumnIndexInRange > (endColumnIndex - startColumnIndex)) {
      Logger.log(`오류: 설정된 날짜 열 '${DATE_COLUMN_LETTER}'(절대 인덱스 ${dateColumnIndexAbsolute})이(가) 데이터 범위 '${TASK_DATA_RANGE}'(절대 인덱스 ${startColumnIndex}~${endColumnIndex}) 내에 없습니다.`);
      SpreadsheetApp.getUi().alert(`오류: 설정된 날짜 열 '${DATE_COLUMN_LETTER}'이(가) 데이터 범위 '${TASK_DATA_RANGE}' 내에 없습니다.`);
      return;
  }
  Logger.log(`데이터 범위 '${TASK_DATA_RANGE}'에서 날짜 열 '${DATE_COLUMN_LETTER}'은(는) 가져온 데이터 배열의 인덱스 ${dateColumnIndexInRange}에 해당합니다.`);

  // 데이터 가져오기
  
  const taskDataRange = taskSheet.getRange(rangeA1Notation);
  const taskValues = taskDataRange.getValues();
  const headers = taskValues[0]; // 첫 번째 행은 헤더
  const recentTasks = [];

  Logger.log(`${taskValues.length - 1}개의 행 데이터 로드 시도 중... (헤더 제외)`);

  // 데이터 필터링 루프
  for (let i = 1; i < taskValues.length; i++) { // i=0은 헤더이므로 1부터 시작
    const row = taskValues[i];
    const dateCellValue = row[dateColumnIndexInRange]; // 계산된 인덱스 사용 (E열 = 인덱스 1)

    let matched = false; // 현재 행이 조건에 맞는지 여부
    let taskDate = null; // 파싱된 날짜를 저장할 변수

    // --- 날짜 값 타입 및 형식 확인 시작 ---
    try {
      // 1. Date 객체인지 확인
      if (dateCellValue instanceof Date) {
        taskDate = new Date(dateCellValue);
      }
      // 2. 문자열인지 확인
      else if (typeof dateCellValue === 'string' && dateCellValue.trim() !== '') {
        const dateString = dateCellValue.trim();
        let parsedSuccessfully = false;

        // 2a. yyyy. M. d 형식 (점과 공백 허용) 파싱 시도 <--- 이 부분 추가됨!
        const partsYMD_dotSpace = dateString.match(/^(\d{4})\.?\s*(\d{1,2})\.?\s*(\d{1,2})\.?$/);
        if (partsYMD_dotSpace) {
          const year = parseInt(partsYMD_dotSpace[1]);
          const month = parseInt(partsYMD_dotSpace[2]) - 1; // Month is 0-indexed
          const day = parseInt(partsYMD_dotSpace[3]);
          // 간단한 유효성 검사
          if (year > 1900 && month >= 0 && month < 12 && day > 0 && day <= 31) {
            taskDate = new Date(year, month, day);
            parsedSuccessfully = true;
            // Logger.log(`Row ${i+1}: Parsed '${dateString}' as yyyy. M. d`); // 디버깅 필요시 주석 해제
          }
        }

        // 2b. yyyy-MM-dd 형식 파싱 시도
        if (!parsedSuccessfully) {
          const partsYMD_hyphen = dateString.match(/^(\d{4})-(\d{1,2})-(\d{1,2})$/);
          if (partsYMD_hyphen) {
            const year = parseInt(partsYMD_hyphen[1]);
            const month = parseInt(partsYMD_hyphen[2]) - 1;
            const day = parseInt(partsYMD_hyphen[3]);
            if (year > 1900 && month >= 0 && month < 12 && day > 0 && day <= 31) {
                taskDate = new Date(year, month, day);
                parsedSuccessfully = true;
                // Logger.log(`Row ${i+1}: Parsed '${dateString}' as yyyy-MM-dd`);
            }
          }
        }

        // 2c. 다른 표준 형식 시도 (new Date() 활용)
        if (!parsedSuccessfully) {
          const potentiallyParsedDate = new Date(dateString);
          if (potentiallyParsedDate instanceof Date && !isNaN(potentiallyParsedDate.getTime())) {
            if (potentiallyParsedDate.getFullYear() > 1900) {
                taskDate = potentiallyParsedDate;
                // Logger.log(`Row ${i+1}: Parsed '${dateString}' using new Date()`);
            }
          }
        }
      }
      // 3. 숫자인지 확인 (Google Sheets 날짜 일련번호 처리 시도)
      else if (typeof dateCellValue === 'number' && dateCellValue > 0) {
          // 참고: 숫자 형식 날짜는 시간대 문제 등으로 정확도가 떨어질 수 있음
          const formattedDateStr = Utilities.formatDate(new Date(1899, 11, 30 + dateCellValue), timeZone, "yyyy-MM-dd");
          taskDate = new Date(formattedDateStr); // yyyy-MM-dd 형식으로 재파싱
          // Logger.log(`Row ${i+1}: Parsed number ${dateCellValue} to ${formattedDateStr}`);
      }
      // --- 날짜 값 타입 및 형식 확인 끝 ---


      // --- 파싱된 날짜로 최종 비교 ---
      if (taskDate instanceof Date && !isNaN(taskDate.getTime())) {
        taskDate.setHours(0, 0, 0, 0); // 시간 부분 초기화
        if (taskDate.getTime() >= pastDateTimestamp) {
          matched = true;
        }
        // 로그 추가하여 타임스탬프 비교 확인 (디버깅 시 유용)
        // Logger.log(`Row ${i+1}: Comparing TaskDate(${Utilities.formatDate(taskDate, timeZone, "yyyy-MM-dd HH:mm:ss")} / ${taskDate.getTime()}) >= PastDate(${Utilities.formatDate(pastDate, timeZone, "yyyy-MM-dd HH:mm:ss")} / ${pastDateTimestamp}) -> Matched: ${matched}`);
      } else {
         // 날짜로 인식/파싱되지 않은 경우 로그 (디버깅 시 유용)
         // if (dateCellValue) { // 빈 셀이 아닌 경우만 로그
         //    Logger.log(`Row ${i+1}: Could not parse date value '${dateCellValue}' (Type: ${typeof dateCellValue})`);
         // }
      }
       // --- 최종 비교 끝 ---

    } catch(e) {
        Logger.log(`Row ${i+1}: Error processing date value '${dateCellValue}'. Error: ${e}`);
    }


    // 조건(날짜)에 맞고, 빈 행이 아닐 경우 추가
    if (matched && row.some(cell => cell !== "")) {
      recentTasks.push(row);
      // Logger.log(`Row ${i+1}: Added to recentTasks.`); // 디버깅 필요시 주석 해제
    }
  } // End of for loop

  Logger.log(`최종 필터링 결과: 최근 ${DAYS_TO_LOOK_BACK}일 동안의 작업 ${recentTasks.length}건을 찾았습니다.`);

  // 분석할 데이터가 없는 경우 처리
  if (recentTasks.length === 0) {
    Logger.log("분석할 최근 작업 데이터가 없습니다. advice 시트에 기록합니다.");
    const nextRow = adviceSheet.getLastRow() + 1;
    adviceSheet.getRange(nextRow, 1, 1, 3).setValues([[currentDate, "분석할 최근 작업 데이터 없음", "분석할 최근 작업 데이터 없음"]]);
    // 사용자에게 알림을 줄 수도 있음
    // SpreadsheetApp.getUi().alert("지난 7일간의 작업 데이터가 없어 분석을 건너<0xEB><0x9B><0x81>니다.");
    return; // 데이터 없으면 여기서 종료
  }

  // --- Gemini API 호출을 위한 데이터 형식화 ---
  let taskDataString = `최근 ${DAYS_TO_LOOK_BACK}일간의 작업 내역 (총 ${recentTasks.length}건):\n`;
  // 헤더 추가 (가져온 데이터 기준)
  taskDataString += headers.join(", ") + "\n";
  // 데이터 행 추가
  recentTasks.forEach(task => {
    const rowString = task.map(cell => {
      if (cell instanceof Date) {
        // 날짜 객체는 yyyy-MM-dd 형식으로 변환
        return Utilities.formatDate(cell, timeZone, "yyyy-MM-dd");
      }
      // null, undefined는 빈 문자열로, 나머지는 문자열로 변환
      return cell === null || cell === undefined ? "" : String(cell);
    }).join(", "); // 각 셀은 쉼표와 공백으로 구분
    taskDataString += rowString + "\n"; // 각 행은 줄바꿈으로 구분
  });

  // 너무 긴 데이터 문자열 로깅은 성능에 영향 줄 수 있으므로 필요시 주석 해제
  // Logger.log("Gemini에게 전달될 작업 데이터 문자열:\n" + taskDataString);

  // --- Gemini API 호출 함수 정의 ---
  function callGeminiAPI(prompt, taskData) {
    const url = `https://generativelanguage.googleapis.com/v1beta/models/${GEMINI_MODEL}:generateContent?key=${GEMINI_API_KEY}`;

    const payload = {
      "contents": [
        {
          "parts": [ { "text": prompt + "\n\n다음은 분석해야 할 최근 작업 데이터입니다:\n" + taskData } ]
        }
      ],
       "generationConfig": { // 필요한 경우 생성 옵션 추가
         "temperature": 0.7, // 창의성 조절 (0~1)
         "maxOutputTokens": 1500 // 최대 출력 토큰 수 조절
       },
       "safetySettings": [ // 안전 설정 (필요시 조절)
            { "category": "HARM_CATEGORY_HARASSMENT", "threshold": "BLOCK_MEDIUM_AND_ABOVE" },
            { "category": "HARM_CATEGORY_HATE_SPEECH", "threshold": "BLOCK_MEDIUM_AND_ABOVE" },
            { "category": "HARM_CATEGORY_SEXUALLY_EXPLICIT", "threshold": "BLOCK_MEDIUM_AND_ABOVE" },
            { "category": "HARM_CATEGORY_DANGEROUS_CONTENT", "threshold": "BLOCK_MEDIUM_AND_ABOVE" }
       ]
    };

    const options = {
      'method': 'post',
      'contentType': 'application/json',
      'payload': JSON.stringify(payload),
      'muteHttpExceptions': true // API 오류 시 스크립트 중단 대신 응답 받기
    };

    try {
      Logger.log(`Gemini API 호출 시작 (프롬프트: ${prompt.substring(0, 20)}...)`);
      const response = UrlFetchApp.fetch(url, options);
      const responseCode = response.getResponseCode();
      const responseBody = response.getContentText();

      if (responseCode === 200) {
        const jsonResponse = JSON.parse(responseBody);
        // 응답 구조 확인 및 텍스트 추출
        if (jsonResponse.candidates && jsonResponse.candidates.length > 0 &&
            jsonResponse.candidates[0].content && jsonResponse.candidates[0].content.parts &&
            jsonResponse.candidates[0].content.parts.length > 0 && jsonResponse.candidates[0].content.parts[0].text) {
          Logger.log(`Gemini API 응답 성공 (프롬프트: ${prompt.substring(0, 20)}...)`);
          return jsonResponse.candidates[0].content.parts[0].text;
        } else if (jsonResponse.candidates && jsonResponse.candidates[0] && jsonResponse.candidates[0].finishReason) {
           // finishReason이 SAFETY 등이면 content가 없을 수 있음
           Logger.log(`Gemini API 응답 완료되었으나 내용 없음 (프롬프트: ${prompt.substring(0, 20)}...). 이유: ${jsonResponse.candidates[0].finishReason}, 응답: ${responseBody}`);
           return `API 응답 내용 없음 (이유: ${jsonResponse.candidates[0].finishReason})`;
        } else {
          Logger.log(`Gemini API 응답 형식 오류 (프롬프트: ${prompt.substring(0, 20)}...). 응답: ${responseBody}`);
          return `API 응답 형식 오류`;
        }
      } else {
        Logger.log(`Gemini API 오류 (프롬프트: ${prompt.substring(0, 20)}...). 응답 코드: ${responseCode}, 내용: ${responseBody}`);
        // 오류 메시지에서 더 유용한 정보 추출 시도
         try {
            const errorResponse = JSON.parse(responseBody);
            if (errorResponse.error && errorResponse.error.message) {
                return `API 오류 (${responseCode}): ${errorResponse.error.message}`;
            }
         } catch (e) { /* JSON 파싱 실패 시 원래 오류 메시지 사용 */ }
        return `API 오류 (${responseCode})`;
      }
    } catch (error) {
      Logger.log(`Gemini API 호출 중 예외 발생 (프롬프트: ${prompt.substring(0, 20)}...): ${error}`);
      return `API 호출 예외: ${error.message}`;
    }
  } // --- End of callGeminiAPI function ---

  // --- 각 프롬프트에 대해 API 호출 실행 ---
  const secretaryAdvice = callGeminiAPI(secretaryPrompt, taskDataString);
  Utilities.sleep(1000); // API 호출 사이에 약간의 지연 시간 추가 (API Rate Limit 방지 목적)
  const nutritionistAdvice = callGeminiAPI(nutritionistPrompt, taskDataString);

  // --- 결과를 advice 시트에 기록 ---
  const nextRow = adviceSheet.getLastRow() + 1;
  // 결과를 A열(날짜), B열(비서 조언), C열(영양사 조언)에 기록
  try {
      adviceSheet.getRange(nextRow, 1, 1, 3).setValues([[currentDate, secretaryAdvice, nutritionistAdvice]]);
      Logger.log(`결과를 ${ADVICE_SHEET_NAME} 시트 ${nextRow}행에 기록했습니다.`);

  } catch (e) {
      Logger.log(`결과를 ${ADVICE_SHEET_NAME} 시트에 기록하는 중 오류 발생: ${e}`);

  }

} // --- End of triggerGeminiUpdate function ---
