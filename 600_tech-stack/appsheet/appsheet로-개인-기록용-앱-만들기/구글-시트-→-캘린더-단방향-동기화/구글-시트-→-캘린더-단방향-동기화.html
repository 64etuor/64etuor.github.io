구글 시트 → 캘린더 단방향 동기화

구글 캘린더 서비스 API가 아닌 CalendarApp 이라는 앱스크립트에서 기본적으로 제공하는 클래스를 호출하므로 서비스를 추가할 필요는 없습니다.
이 코드에는 캘린더의 이벤트를 삭제하는 로직이 담겨 있습니다.

로직을 제대로 이해하시지 않았다면 사용하지 말 것을 권장 드립니다.

본인의 필요에 맞게 코드를 작성해서 적용해야 합니다!!

코드 로직

  • 단방향은 “캘린더 → 시트”와 “시트 → 캘린더”를 분리된 두 함수로 구현
  • 삭제 로직은 “스크립트가 원래 관리하던 이벤트”만(Property에 저장) → “시트에서 빠진 경우”에만 적용
  • 생성/업데이트는 “신규 vs 수정” 분기, 타임스탬프 기반 증분 처리
  • API 과부하 방지를 위해 배치 사이에 sleep() 삽입

AppsScript JS 코드 설정

/************************************************************
 * 1. 상수 및 설정 정의
 ************************************************************/
// 대상 시트 이름
var TARGET_SHEET_NAME = "calendarEvents";

// 데이터 추출할 캘린더 ID 목록 (구글 캘린더 -> 시트 방향)
var SYNC_CALENDAR_IDS = [
  "메인 캘린더 ID", // 메인 캘린더 ID #######################\#본인에 맞게 수정###############################
  "ko.south_korea\#holiday@group.v.calendar.google.com" // 공휴일 캘린더 ID #######################\#본인에 맞게 수정###############################
];

// 동기화 작업을 수행할 메인 캘린더 ID (시트 -> 구글 캘린더 방향)
var MAIN_SYNC_CALENDAR_ID = "동기화될 캘린더"; // #######################\#본인에 맞게 수정###############################
var HOLIDAY_CALENDAR_ID = "ko.south_korea\#holiday@group.v.calendar.google.com"; // 동기화 시 제외할 ID (읽기 전용 등)

// 시트 열 인덱스 상수 (0부터 시작)
var COL_EVENT_ID = 0;
var COL_TITLE = 1;
var COL_START_TIME = 2;
var COL_END_TIME = 3;
var COL_DURATION_MIN = 4; // 소요 시간 (분 단위 정수)
var COL_LOCATION = 5;
var COL_CREATOR = 6;
var COL_ATTENDEES = 7;
var COL_MEET_LINK = 8;
var COL_DESCRIPTION = 9;
var COL_CALENDAR_ID = 10;
var COL_LAST_MODIFIED = 11;

// PropertiesService 키
var LAST_SYNC_TIME_PROP_KEY = 'LAST_CALENDAR_SYNC_TIME';

// API 요청 제한 설정
var BATCH_REQUEST_SIZE = 5; // 한 번에 처리할 요청 수
var WAIT_TIME_MS = 5000;    // 배치 처리 후 대기 시간 (밀리초)

/************************************************************
 * 2. Google Calendar → Google Sheets: 선택된 캘린더 동기화
 ************************************************************/
function exportSelectedCalendars() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(TARGET_SHEET_NAME);

  if (!sheet) {
    sheet = ss.insertSheet(TARGET_SHEET_NAME);
    Logger.log("Sheet '" + TARGET_SHEET_NAME + "' created.");
  }

  sheet.clearContents();
  // 헤더 설정
  var headers = ["Event ID", "Title", "Start", "End", "Duration_m", "Location", "Creator Email", "Attendees", "Meet Link", "Description", "Calendar ID", "Last Modified"];
  sheet.getRange(1, 1, 1, headers.length).setValues([headers]).setFontWeight("bold");

  // 날짜/시간 및 숫자 형식 설정 (헤더 제외)
  var dataRows = sheet.getMaxRows() - 1;
  if (dataRows > 0) {
      sheet.getRange(2, COL_START_TIME + 1, dataRows, 1).setNumberFormat("yyyy-mm-dd hh:mm:ss");
      sheet.getRange(2, COL_END_TIME + 1, dataRows, 1).setNumberFormat("yyyy-mm-dd hh:mm:ss");
      sheet.getRange(2, COL_DURATION_MIN + 1, dataRows, 1).setNumberFormat("0"); // 분 단위 정수
      sheet.getRange(2, COL_LAST_MODIFIED + 1, dataRows, 1).setNumberFormat("yyyy-mm-dd hh:mm:ss");
  }


  // 가져올 날짜 범위 지정
  var startDate = new Date(2023, 0, 1); // 데이터 가져올 MIN 날짜 #######################\#본인에 맞게 수정###############################
  var endDate = new Date(2030, 0, 1);   // 데이터 가져올 MAX 날짜 #######################\#본인에 맞게 수정###############################

  var allEventData = [];

  SYNC_CALENDAR_IDS.forEach(function(calendarId) {
    var calendar = CalendarApp.getCalendarById(calendarId);
    if (!calendar) {
      Logger.log("Calendar not found: " + calendarId + ". Skipping.");
      return;
    }
    Logger.log("Processing calendar: " + calendar.getName() + " (" + calendarId + ")");

    try {
      var events = calendar.getEvents(startDate, endDate);
      Logger.log("Found " + events.length + " events in calendar: " + calendarId);

      events.forEach(function(event) {
          var startTime = event.getStartTime();
          var endTime = event.getEndTime();

          // 소요 시간 계산 (분 단위 정수)
          var durationMinutes = 0; // 기본값
          if (startTime && endTime) {
              var diffMs = endTime.getTime() - startTime.getTime();
              if (diffMs > 0) {
                  // 밀리초 차이를 분으로 변환하고 반올림
                  durationMinutes = Math.round(diffMs / (1000 * 60));
              }
          }

          var attendees = event.getGuestList().map(function(guest) { return guest.getEmail(); }).join(", ");
          var meetLink = ""; try { var conferenceData = event.getConferenceData(); if (conferenceData && conferenceData.getEntryPoints().length > 0) { var videoEntryPoint = conferenceData.getEntryPoints().find(function(entryPoint){ return entryPoint.getEntryPointType() == CalendarApp.ConferenceEntryPointType.VIDEO; }); if(videoEntryPoint){ meetLink = videoEntryPoint.getUri(); } } } catch (e) {}
          var creatorEmail = "Unknown"; try { var creators = event.getCreators(); if (creators && creators.length > 0) { creatorEmail = creators[0].getEmail ? creators[0].getEmail() : creators[0]; } } catch (e) { Logger.log("Could not get creator for event: " + event.getTitle() + ". Error: " + e.message); }

          // allEventData 배열에 데이터 추가
          allEventData.push([
            event.getId(), event.getTitle(), startTime, endTime,
            durationMinutes, // 분 단위 정수 추가
            event.getLocation() || "", creatorEmail, attendees || "", meetLink || "",
            event.getDescription() || "", calendarId, "" // Last Modified 는 비워둠 (동기화 시 업데이트)
          ]);
      });
    } catch (e) {
      Logger.log("Error fetching events for calendar " + calendarId + ": " + e.message);
    }
  });

  if (allEventData.length > 0) {
    sheet.getRange(2, 1, allEventData.length, headers.length).setValues(allEventData);
    Logger.log(allEventData.length + " events exported successfully to sheet '" + TARGET_SHEET_NAME + "'.");
  } else {
    Logger.log("No events found in the specified date range for the selected calendars.");
  }

  sheet.autoResizeColumns(1, headers.length);
}


/************************************************************
 * 3. Google Sheets → Google Calendar 동기화 (Last Modified 기준)
 ************************************************************/
function syncSheetToCalendar() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(TARGET_SHEET_NAME);

  if (!sheet) {
    Logger.log("Sheet '" + TARGET_SHEET_NAME + "' not found. Exiting function.");
    return;
  }

  var mainCalendar = CalendarApp.getCalendarById(MAIN_SYNC_CALENDAR_ID);
  if (!mainCalendar) {
    Logger.log("Main calendar (" + MAIN_SYNC_CALENDAR_ID + ") not found. Exiting function.");
    return;
  }
  Logger.log("Starting sync from sheet '" + TARGET_SHEET_NAME + "' to calendar '" + mainCalendar.getName() + "' based on last modified time.");

  // --- 마지막 동기화 시간 가져오기 ---
  var scriptProperties = PropertiesService.getScriptProperties();
  var lastSyncTimeStr = scriptProperties.getProperty(LAST_SYNC_TIME_PROP_KEY);
  var lastSyncTime = lastSyncTimeStr ? new Date(parseInt(lastSyncTimeStr)) : new Date(0); // 없으면 Epoch (1970-01-01)
  Logger.log("Last successful sync completed around: " + (lastSyncTime.getTime() === 0 ? "Never" : lastSyncTime.toISOString()));
  var currentSyncStartTime = new Date(); // 현재 동기화 시작 시간 기록

  // 시트 데이터 읽기 (헤더 제외)
  var dataRange = sheet.getDataRange();
  var sheetData = dataRange.getValues();
  var headerRow = sheetData[0]; // 헤더 저장
  var sheetEvents = sheetData.slice(1); // 데이터 행만 추출

  // --- 시트에 있는 '메인 캘린더 소속' 이벤트 ID 수집 (삭제 로직용) ---
  // *** 수정된 로직 적용 ***
  var sheetEventIds = new Set();
  sheetEvents.forEach(function(row) {
    var eventId = row[COL_EVENT_ID];
    var sourceCalendarId = row[COL_CALENDAR_ID];

    // 유효한 Event ID 형식인지 기본적인 확인
    if (eventId && typeof eventId === 'string' && eventId.includes('@')) {
      // 시트에서 해당 이벤트가 MAIN_SYNC_CALENDAR_ID 소속이거나,
      // sourceCalendarId가 비어있는 경우 (새로 수동 추가된 행으로 간주)에만 추가
      if (sourceCalendarId === MAIN_SYNC_CALENDAR_ID || !sourceCalendarId) {
        sheetEventIds.add(eventId);
      } else {
         // (선택적 로깅) 명시적으로 다른 캘린더 소속이라 제외되는 경우
         // Logger.log("Skipping Event ID '" + eventId + "' from sheet scan for deletion check, because its sourceCalendarId ('" + sourceCalendarId + "') is explicitly set to a different calendar and not the MAIN_SYNC_CALENDAR_ID ('" + MAIN_SYNC_CALENDAR_ID + "').");
      }
    }
  });
  Logger.log("Found " + sheetEventIds.size + " unique event IDs in the sheet considered belonging to the main calendar ('" + MAIN_SYNC_CALENDAR_ID + "' or blank source).");

  // --- 캘린더에서 삭제할 이벤트 식별 및 삭제 ---
  // 동기화 범위 설정 (캘린더에서 이벤트를 가져와 비교할 범위)
  var syncStartDate = new Date(2024, 0, 1); // #######################\#본인에 맞게 수정###############################
  var syncEndDate = new Date(2026, 0, 1);   // #######################\#본인에 맞게 수정###############################
  var calendarEvents;
  try {
      calendarEvents = mainCalendar.getEvents(syncStartDate, syncEndDate);
  } catch (e) {
      Logger.log("Error fetching events from main calendar (" + MAIN_SYNC_CALENDAR_ID + "): " + e.message);
      return; // 캘린더 이벤트 로드 실패 시 중단
  }

  var eventsToDeleteCount = 0;
  calendarEvents.forEach(function(event) {
    var eventId = event.getId();
    // 시트에서 수집한 '메인 캘린더 소속' 이벤트 ID 목록에 없는 경우 삭제 대상
    if (!sheetEventIds.has(eventId)) {
      try {
        // <<< 삭제 전 로그 추가: 어떤 이벤트가 삭제되는지 명확히 확인 >>>
        Logger.log("Deleting event from calendar (ID not found in sheet's list for main calendar): Title='" + event.getTitle() + "', ID='" + eventId + "'");
        event.deleteEvent();
        eventsToDeleteCount++;
        // API 제한 고려하여 일괄 처리 후 대기
        if (eventsToDeleteCount % BATCH_REQUEST_SIZE === 0) {
            Utilities.sleep(WAIT_TIME_MS);
        }
      } catch (e) {
        Logger.log("Error deleting event " + eventId + ": " + e.message + ". Skipping this event.");
      }
    }
  });
  Logger.log("Finished checking for events to delete. Deleted " + eventsToDeleteCount + " events.");


  // --- 시트 데이터를 기반으로 변경된 사항만 캘린더에 생성 또는 업데이트 ---
  var updatedRowCount = 0;
  var createdRowCount = 0;
  var skippedRowCount = 0;
  var requestCount = 0;
  var newEventIdsToUpdate = {}; // 새로 생성된 이벤트 ID를 시트에 다시 쓰기 위한 객체

  for (var i = 0; i < sheetEvents.length; i++) {
    var row = sheetEvents[i];
    var rowIndex = i + 2; // 실제 시트 행 번호 (1-based + 헤더)

    // --- 처리 필요 여부 확인 (Last Modified 기준) ---
    var eventId = row[COL_EVENT_ID];
    var lastModifiedValue = row[COL_LAST_MODIFIED];
    var lastModifiedTime = (lastModifiedValue instanceof Date) ? lastModifiedValue : new Date(0); // 유효한 Date 아니면 Epoch

    // 처리 조건: 1. Event ID가 없는 새 행이거나, 2. 마지막 동기화 시간 이후 수정된 행
    var isNewRow = !eventId || !(typeof eventId === 'string' && eventId.includes('@'));
    var isModifiedSinceLastSync = lastModifiedTime.getTime() > 0 && lastModifiedTime.getTime() > lastSyncTime.getTime();

    if (!isNewRow && !isModifiedSinceLastSync) {
        skippedRowCount++;
        continue; // 변경 없으면 다음 행으로
    }
    // --- 처리 필요 여부 확인 끝 ---

    // 필수 데이터 유효성 검사
    var title = row[COL_TITLE];
    var start = new Date(row[COL_START_TIME]); // 날짜 변환 시도
    var end = new Date(row[COL_END_TIME]);     // 날짜 변환 시도
    if (!title || isNaN(start.getTime()) || isNaN(end.getTime())) {
      Logger.log("Skipping row " + rowIndex + ": Invalid title, start time, or end time. Check data and format.");
      skippedRowCount++;
      continue;
    }
    if (end.getTime() < start.getTime()) {
        Logger.log("Skipping row " + rowIndex + ": End time is before start time.");
        skippedRowCount++;
        continue;
    }

    // 동기화 제외 캘린더 (예: 공휴일) 데이터 건너뛰기
    var sourceCalendarId = row[COL_CALENDAR_ID];
    if (sourceCalendarId && sourceCalendarId === HOLIDAY_CALENDAR_ID) {
      Logger.log("Skipping row " + rowIndex + " (Title: " + title + ") because its sourceCalendarId is set to the read-only HOLIDAY_CALENDAR_ID.");
      skippedRowCount++;
      continue;
    }

    // 이벤트 데이터 추출 (Duration 열은 동기화에 직접 사용 안 함)
    var location = row[COL_LOCATION] || "";
    var description = row[COL_DESCRIPTION] || "";

    try {
      var event = null;
      if (!isNewRow) { // Event ID가 있는 경우, 캘린더에서 해당 이벤트 찾아 시도
          try{
              event = mainCalendar.getEventById(eventId);
          } catch(e) {
              Logger.log("Could not retrieve event with ID '" + eventId + "' for row " + rowIndex + ". Maybe it was deleted or ID is invalid? Error: " + e.message);
              event = null; // 찾지 못하면 null 상태 유지 (아래에서 생성 로직으로 이어짐)
          }
      }

      // API 요청 수 증가 (실제 요청 발생 시점에 가깝게)
      requestCount++;

      if (event) { // 이벤트가 캘린더에 존재 -> 업데이트
        // 변경 사항이 있는지 확인 후 업데이트 (선택적 최적화)
        // 예: if (event.getTitle() !== title || event.getStartTime().getTime() !== start.getTime() || ...) { ... }
        // 여기서는 무조건 업데이트 실행
        event.setTitle(title);
        event.setTime(start, end);
        event.setLocation(location);
        event.setDescription(description);
        updatedRowCount++;
        // 업데이트 시 시트의 Last Modified 시간도 현재 동기화 시간으로 업데이트
        sheet.getRange(rowIndex, COL_LAST_MODIFIED + 1).setValue(currentSyncStartTime);
        Logger.log("Updated event in calendar for row " + rowIndex + " (Title: " + title + ")");

      } else { // 이벤트가 없거나 새 행 -> 생성
        var newEvent = mainCalendar.createEvent(title, start, end, {
          location: location,
          description: description
        });
        var newEventId = newEvent.getId();
        newEventIdsToUpdate[rowIndex] = newEventId; // 나중에 시트에 ID 일괄 업데이트

        // 생성 시 시트의 Calendar ID 열 업데이트 (메인 캘린더 ID로) - 비어있거나 다른 ID였을 경우
        if(sheet.getRange(rowIndex, COL_CALENDAR_ID + 1).getValue() !== MAIN_SYNC_CALENDAR_ID){
            sheet.getRange(rowIndex, COL_CALENDAR_ID + 1).setValue(MAIN_SYNC_CALENDAR_ID);
        }
        // 생성 시 시트의 Last Modified 시간도 현재 동기화 시간으로 업데이트
        sheet.getRange(rowIndex, COL_LAST_MODIFIED + 1).setValue(currentSyncStartTime);

        createdRowCount++;
        Logger.log("Created new event in calendar for row " + rowIndex + " (Title: " + title + "), New Event ID: " + newEventId);
      }

    } catch (e) {
      Logger.log("Error processing row " + rowIndex + " (Title: " + title +"): " + e.message);
      // 오류 발생 시 requestCount를 증가시키지 않거나, 별도 처리 필요
    }

    // API 요청 속도 제한
    if (requestCount >= BATCH_REQUEST_SIZE) {
      Logger.log("Processed " + requestCount + " updates/creations batch, waiting for " + WAIT_TIME_MS / 1000 + " seconds...");
      Utilities.sleep(WAIT_TIME_MS);
      requestCount = 0; // 카운터 리셋
    }
  } // end of loop

  // --- 새로 생성된 이벤트 ID를 시트에 일괄 업데이트 ---
  var rowsToUpdate = Object.keys(newEventIdsToUpdate);
  if (rowsToUpdate.length > 0) {
    Logger.log("Updating sheet with " + rowsToUpdate.length + " new event IDs.");
    rowsToUpdate.forEach(function(rowIndexStr) {
        var rowIndex = parseInt(rowIndexStr, 10);
        try {
            sheet.getRange(rowIndex, COL_EVENT_ID + 1).setValue(newEventIdsToUpdate[rowIndex]);
        } catch (e) {
            Logger.log("Error updating Event ID on sheet for row " + rowIndex + ": " + e.message);
        }
    });
  }

  // --- 동기화 성공 시 마지막 동기화 시간 업데이트 ---
  // 에러 발생 여부와 관계없이 마지막 시도 시간을 기록할지, 아니면 완전 성공 시에만 기록할지 결정 필요
  // 현재 로직은 루프가 끝나면 무조건 업데이트 시도
  try {
      scriptProperties.setProperty(LAST_SYNC_TIME_PROP_KEY, currentSyncStartTime.getTime().toString());
      Logger.log("Successfully updated last sync time marker to: " + currentSyncStartTime.toISOString());
  } catch (e) {
      Logger.log("Error saving last sync time: " + e.message);
  }

  Logger.log("Sync from sheet to calendar finished. Updated: " + updatedRowCount + ", Created: " + createdRowCount + ", Skipped: " + skippedRowCount + " rows.");
}


/************************************************************
 * 4. 모든 캘린더 목록 출력
 ************************************************************/
function listAllCalendars() {
  try {
    var calendars = CalendarApp.getAllCalendars();

    if (!calendars || calendars.length === 0) {
      Logger.log("No calendars found for this user.");
      return;
    }

    Logger.log("List of Your Google Calendars:");
    calendars.forEach(function(calendar) {
      Logger.log(" - Name: " + calendar.getName() + ", ID: " + calendar.getId() + ", Primary: " + calendar.isMyPrimaryCalendar() + ", Owned: " + calendar.isOwnedByMe() + ", TimeZone: " + calendar.getTimeZone());
    });
  } catch (e) {
    Logger.log("Error listing calendars: " + e.message);
  }
}

  SpreadsheetApp.getUi()

      .addSeparator()

      .addSeparator()
