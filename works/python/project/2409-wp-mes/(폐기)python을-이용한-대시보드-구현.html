(폐기)Python을 이용한 대시보드 구현

대시보드 구현을 위한 주요 단계

  1. 날짜 필터링 기능: 연도, 월, 주차, 날짜 필터링을 추가.
  2. 생산 차트: 생산수량, 불량수량을 누적 막대 그래프로 표시.
  3. 불량율 그래프: 불량율을 선 그래프로 표시.
  4. 프리히터 온도 기반 Cpk 그래프: 공정 파라미터의 변화를 감지하는 그래프를 cpk로 표시.
  5. 데이터베이스 실시간 업데이트: 1분마다 데이터베이스에서 새로운 데이터 확인 후 업데이트.
  6. 2열로 차트 배치: 한 화면에 2열로 차트를 배치.

이제 단계별로 코드를 작성합니다:

1. 기본 대시보드 레이아웃 및 데이터베이스 연결

import numpy as np
import plotly.graph_objects as go
from scipy.stats import norm
from sqlalchemy import create_engine
import urllib.parse
import pandas as pd
from dash import dcc, html
import dash
from dash.dependencies import Input, Output
import time

# 데이터베이스 접속 정보 설정
username = urllib.parse.quote_plus('test')
password = urllib.parse.quote_plus('0908')
host = 'localhost'
port = '5432'
database = 'postgres'

DATABASE_URL = f"postgresql://{username}:{password}@{host}:{port}/{database}"

engine = create_engine(DATABASE_URL)

# 대시보드 설정
app = dash.Dash(__name__)

# 실시간 데이터 업데이트 함수
def load_data():
    query = "SELECT * FROM production_data"  # 테이블의 이름
    df = pd.read_sql(query, engine)
    return df

# 레이아웃 설정
app.layout = html.Div([

    dcc.DatePickerRange(
        id='date-picker',

        display_format='YYYY-MM-DD'
    ),
    
    # 차트들
    html.Div([
        dcc.Graph(id='production-chart'),
        dcc.Graph(id='defect-rate-chart')
    ], style={'display': 'flex', 'flex-direction': 'row'}),
    
    html.Div([
        dcc.Graph(id='cpk-chart')
    ], style={'display': 'flex', 'flex-direction': 'row'})
])

# 데이터 업데이트 및 차트 생성 콜백
@app.callback(
    [Output('production-chart', 'figure'),
     Output('defect-rate-chart', 'figure'),
     Output('cpk-chart', 'figure')],
    [Input('date-picker', 'start_date'),
     Input('date-picker', 'end_date')]
)
def update_charts(start_date, end_date):
    # 데이터 불러오기
    df = load_data()

    # 날짜 필터 적용
    mask = (df['date_of_entry'] >= start_date) & (df['date_of_entry'] <= end_date)
    filtered_df = df.loc[mask]
    
    # 1. 생산 차트
    production_chart = create_production_chart(filtered_df)
    
    # 2. 불량율 차트
    defect_rate_chart = create_defect_rate_chart(filtered_df)
    
    # 3. Cpk 차트
    cpk_chart = create_cpk_chart(filtered_df)
    
    return production_chart, defect_rate_chart, cpk_chart

2. 차트 생성 함수들

(1) 생산 차트

def create_production_chart(df):
    df['is_ng'] = df['cell_assy_ok_ng'] == 'NG'
    
    production_summary = df.groupby('date_column').agg(
        production_count=('id', 'count'),
        defect_count=('is_ng', 'sum')
    ).reset_index()
    
    production_summary['good_count'] = production_summary['production_count'] - production_summary['defect_count']
    
    fig = go.Figure()
    fig.add_trace(go.Bar(
        x=production_summary['date_column'],
        y=production_summary['good_count'],
        name='Good',
        marker_color='blue'
    ))
    
    fig.add_trace(go.Bar(
        x=production_summary['date_column'],
        y=production_summary['defect_count'],
        name='NG',
        marker_color='red'
    ))
    
    fig.update_layout(
        title="Production Chart (Good vs NG)",
        barmode='stack'
    )
    
    return fig

(2) 불량율 차트

def create_defect_rate_chart(df):
    df['is_ng'] = df['cell_assy_ok_ng'] == 'NG'
    
    production_summary = df.groupby('date_column').agg(
        production_count=('id', 'count'),
        defect_count=('is_ng', 'sum')
    ).reset_index()
    
    production_summary['defect_rate'] = (production_summary['defect_count'] / 
                                         production_summary['production_count']) * 100
    
    fig = go.Figure()
    fig.add_trace(go.Scatter(
        x=production_summary['date_column'],
        y=production_summary['defect_rate'],
        mode='lines+markers',
        name='불량율'
    ))
    
    fig.update_layout(
        title="불량율 (%)",
        yaxis_title="불량율 (%)"
    )
    
    return fig

(3) Cpk 차트

def create_cpk_chart(df):
    df['temp_diff'] = abs(df['preheater_current_temp'] - df['preheater_setting_temp'])
    
    mean_temp = df['preheater_current_temp'].mean()
    sigma_temp = df['preheater_current_temp'].std()
    
    usl = df['preheater_setting_temp'].mean() * 1.05
    lsl = df['preheater_setting_temp'].mean() * 0.95
    
    fig = go.Figure()
    
    # 분포 곡선
    x = np.linspace(mean_temp - 4 * sigma_temp, mean_temp + 4 * sigma_temp, 100)
    y = norm.pdf(x, mean_temp, sigma_temp)
    
    fig.add_trace(go.Scatter(
        x=x, y=y, mode='lines', name='Current Temperature Distribution',
        line=dict(color='blue')
    ))
    
    # USL, LSL, 중앙값
    fig.add_shape(type='line', x0=lsl, x1=lsl, y0=0, y1=max(y), line=dict(color='red', dash='dash'))
    fig.add_shape(type='line', x0=usl, x1=usl, y0=0, y1=max(y), line=dict(color='red', dash='dash'))
    
    fig.add_shape(type='line', x0=mean_temp, x1=mean_temp, y0=0, y1=max(y), line=dict(color='green', dash='dash'))
    
    fig.update_layout(
        title="Cpk 그래프",
        xaxis_title="프리히터 온도",
        yaxis_title="밀도",
    )
    
    return fig
