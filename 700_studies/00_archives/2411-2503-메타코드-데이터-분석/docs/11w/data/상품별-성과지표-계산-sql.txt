CREATE OR REPLACE TABLE  `datafashion-241006.ga4_241222_test.item_interaction_summary` AS
-- 여러분의 프로젝트명과 데이터셋 이름을 입력해주세요! 
WITH item_events AS (
  SELECT
    DATE(TIMESTAMP_MICROS(event_timestamp)) AS event_date,
    items.item_name,
    event_name,
    COUNT(*) AS event_count
  FROM
    `datafashion-241006.ga4_241222_test.events_*`,
    -- 여러분의 프로젝트명과 데이터셋 이름을 입력해주세요! 
    UNNEST(items) AS items
  WHERE
    _TABLE_SUFFIX BETWEEN '20201201' AND '20201231'
    AND items.item_name IS NOT NULL
    AND event_name IN ('view_item', 'add_to_cart', 'purchase')
  GROUP BY
    event_date,
    items.item_name,
    event_name
),
pivoted AS (
  SELECT
    event_date,
    item_name,
    SUM(CASE WHEN event_name = 'view_item' THEN event_count ELSE 0 END) AS view_count,
    SUM(CASE WHEN event_name = 'add_to_cart' THEN event_count ELSE 0 END) AS add_to_cart_count,
    SUM(CASE WHEN event_name = 'purchase' THEN event_count ELSE 0 END) AS purchase_count
  FROM
    item_events
  GROUP BY
    event_date,
    item_name
)
SELECT
  event_date,
  item_name,
  view_count,
  add_to_cart_count,
  purchase_count
FROM
  pivoted
ORDER BY
  purchase_count DESC;
