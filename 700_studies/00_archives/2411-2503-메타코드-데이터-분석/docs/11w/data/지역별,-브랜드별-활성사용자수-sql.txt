CREATE OR REPLACE TABLE `datafashion-241006.ga4_241222_test.daily_active_users_by_city_and_brand` AS
WITH date_series AS (
  SELECT
    DATE_ADD(DATE '2020-12-01', INTERVAL day_offset DAY) AS target_date
  FROM
    UNNEST(GENERATE_ARRAY(0, 30)) AS day_offset
),
user_metrics AS (
  SELECT
    target_date,
    geo.city AS city,
    items.item_brand AS product_brand,
    COUNT(DISTINCT user_pseudo_id) AS active_users
  FROM
    `datafashion-241006.ga4_241222_test.events_*`,
    UNNEST(items) AS items,  
    date_series
  WHERE
    _TABLE_SUFFIX = FORMAT_DATE('%Y%m%d', target_date)
    AND _TABLE_SUFFIX BETWEEN '20201201' AND '20201231'
  GROUP BY
    target_date,
    city,
    product_brand
)
SELECT 
  FORMAT_DATE('%Y-%m-%d', target_date) AS target_date,
  city,
  product_brand,
  active_users
FROM user_metrics;
