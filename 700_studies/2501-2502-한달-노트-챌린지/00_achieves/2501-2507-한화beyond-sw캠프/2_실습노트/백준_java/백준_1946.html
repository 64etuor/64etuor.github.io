내 코드

/*  
언제나 최고만을 지향하는 굴지의 대기업 진영 주식회사가 신규 사원 채용을 실시한다. 인재 선발 시험은 1차 서류심사와 2차 면접시험으로 이루어진다. 최고만을 지향한다는 기업의 이념에 따라 그들은 최고의 인재들만을 사원으로 선발하고 싶어 한다.  
  
그래서 진영 주식회사는, 다른 모든 지원자와 비교했을 때 서류심사 성적과 면접시험 성적 중 적어도 하나가 다른 지원자보다 떨어지지 않는 자만 선발한다는 원칙을 세웠다. 즉, 어떤 지원자 A의 성적이 다른 어떤 지원자 B의 성적에 비해 서류 심사 결과와 면접 성적이 모두 떨어진다면 A는 결코 선발되지 않는다.  
  
이러한 조건을 만족시키면서, 진영 주식회사가 이번 신규 사원 채용에서 선발할 수 있는 신입사원의 최대 인원수를 구하는 프로그램을 작성하시오.  
  
입력  
첫째 줄에는 테스트 케이스의 개수 T(1 ≤ T ≤ 20)가 주어진다. 각 테스트 케이스의 첫째 줄에 지원자의 숫자 N(1 ≤ N ≤ 100,000)이 주어진다. 둘째 줄부터 N개 줄에는 각각의 지원자의 서류심사 성적, 면접 성적의 순위가 공백을 사이에 두고 한 줄에 주어진다. 두 성적 순위는 모두 1위부터 N위까지 동석차 없이 결정된다고 가정한다.  
  
출력  
각 테스트 케이스에 대해서 진영 주식회사가 선발할 수 있는 신입사원의 최대 인원수를 한 줄에 하나씩 출력한다.  
* */  
  
import java.io.BufferedReader;  
import java.io.IOException;  
import java.io.InputStreamReader;  
import java.util.Arrays;  
import java.util.StringTokenizer;  
  
public class Backjoon_1946 {  
    public static void main(String[] args) throws IOException {  
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));  
        int T = Integer.parseInt(br.readLine());  
          
        for(int i = 0; i < T; i++) {  
            int N = Integer.parseInt(br.readLine());  
            int[][] scores = new int[N][2];  
              
            for(int j = 0; j < N; j++) {  
                StringTokenizer st = new StringTokenizer(br.readLine(), " ");  
                scores[j][0] = Integer.parseInt(st.nextToken());  
                scores[j][1] = Integer.parseInt(st.nextToken());  
            }  
            Arrays.sort(scores, (o1, o2) -> o1[0] - o2[0]);  
              
            int count = 1;  
            int min = scores[0][1];  
            for(int j = 1; j < N; j++) {  
                if(scores[j][1] < min) {  
                    min = scores[j][1];  
                    count++;  
                }  
            }  
            System.out.println(count);
        }  
    }  
}

인터넷 코드

원본 링크 : 루이의 알고리즘

import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.util.ArrayList;
import java.util.Collections;

class Employee implements Comparable<Employee> {
    int paper;
    int interview;

    Employee (int paper, int interview) {
        this.paper = paper;
        this.interview = interview;
    }

    @Override
    public int compareTo(Employee o) {
        // 서류순위를 기준으로 오름차순 정렬
        return this.paper - o.paper;
    }
}

public class Main {
    static int T, N;
    static ArrayList<Employee> list;

    void Solve() {
        int ans = 1; // 서류1등은 일단 무조건 합격임
        int pivot = list.get(0).interview; //서류 1등의 면접점수를 기준점으로 삼음
        for (int i=1; i<N; i++) { // 서류 2등부터 비교 시작
            if (list.get(i).interview < pivot) {
                // 현재 지원자의 면접 순위가 앞의 수위보다 높다면 합격
                ans++; // 합격자수 증가
                pivot = list.get(i).interview; // 기준을 현재 인터뷰 순위로 교체
            }
        }
        System.out.println(ans);
    }
 
    public static void main(String[] args) throws Exception {
        Main m = new Main();
        InputStreamReader reader = new InputStreamReader(System.in);
        BufferedReader br = new BufferedReader(reader);
        
        StringTokenizer st = new StringTokenizer(br.readLine());
        T = Integer.parseInt(st.nextToken());
        for (int i=0; i<T; i++) {
            st = new StringTokenizer(br.readLine());
            N = Integer.parseInt(st.nextToken());
            list = new ArrayList<>();
            for (int j=0; j<N; j++) {
                st = new StringTokenizer(br.readLine());
                int a = Integer.parseInt(st.nextToken());
                int b = Integer.parseInt(st.nextToken());
                list.add(new Employee(a, b));
            }
            Collections.sort(list);//오름차순 정렬
            m.Solve();
        }
    }
}

2️⃣ 코드별 특징 분석

✅ 사용자 코드

int[][] scores = new int[N][2];
...
Arrays.sort(scores, (o1, o2) -> o1[0] - o2[0]);  // 서류 기준 정렬

  • int[][] 2차원 배열 사용

    • 각 지원자의 성적을 배열로 저장 (scores[i][0] = 서류 순위, scores[i][1] = 면접 순위)
    • 배열을 첫 번째 값(서류 순위) 기준으로 정렬

  • 기준 점수(min)로 합격 여부 판단

int min = scores[0][1];  // 서류 1등의 면접 성적
for(int j = 1; j < N; j++) {
    if(scores[j][1] < min) {  // 더 높은 면접 성적을 가진 사람만 추가
        min = scores[j][1];
        count++;
    }
}
  • 첫 번째 지원자의 면접 점수를 min으로 설정
  • 이후, 더 좋은 면접 성적(순위가 낮은 값)을 가진 지원자만 선발

✔️ 장점

  • 구조가 단순하고 직관적
  • 배열을 사용하므로 메모리 사용량이 적음

❌ 단점

  • 가독성이 낮음 → scores[i][0], scores[i][1]처럼 숫자로 접근해야 함
  • 객체지향적이지 않음 → 데이터를 명확히 표현하는 클래스가 없음

✅ 인터넷 코드

static ArrayList<Employee> list;
...
Collections.sort(list);  // Comparable을 활용한 정렬
  • ArrayList<Employee> 사용
    • Employee 클래스에 paper, interview를 저장해 객체 지향적으로 접근 가능
    • Comparable<Employee>을 구현해 정렬 기준을 설정 
@Override
public int compareTo(Employee o) {
    return this.paper - o.paper;  // 서류 기준 오름차순 정렬
}
  • Collections.sort(list) 호출 시, compareTo() 메서드가 실행되어 자동 정렬됨
  • 서류 성적이 낮을수록(좋을수록) 앞으로 정렬됨
void Solve() {
    int ans = 1;
    int pivot = list.get(0).interview;
    for (int i = 1; i < N; i++) {
        if (list.get(i).interview < pivot) {  // 더 높은 면접 순위만 선발
            ans++;
            pivot = list.get(i).interview;
        }
    }
    System.out.println(ans);
}
  • 첫 번째 지원자의 면접 순위를 pivot으로 설정
  • 이후 면접 성적이 더 높은(순위가 낮은) 지원자만 선발

✔️ 장점

  • 객체지향적 접근 → Employee 클래스가 있어서 데이터 의미가 명확
  • 가독성이 좋음 → list.get(i).interview처럼 직관적으로 필드 접근 가능
  • 확장성이 좋음 → compareTo() 수정 시 정렬 기준을 쉽게 변경할 수 있음

❌ 단점

  • 배열보다 메모리를 조금 더 사용
  • 코드가 약간 더 길어짐 (객체 생성 필요)

3️⃣ 성능 비교

사용자 코드 (배열) 인터넷 코드 (객체 리스트)
시간 복잡도 O(N log N) O(N log N)
정렬 속도 Arrays.sort() (빠름) Collections.sort() (Comparable 활용)
메모리 사용량 낮음 (int[][]) 높음 (ArrayList<Employee>)

둘 다 O(N log N) 정렬 후 O(N) 탐색이므로 성능 차이는 크지 않음.
하지만 객체 리스트는 추가적인 메모리 사용이 발생할 수 있음.
